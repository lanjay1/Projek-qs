// ============================================================
// ISR -- SENSOR PULSA
// Hanya micros(), counter, flag. Tidak ada Serial, malloc, dll.
// ============================================================
void IRAM_ATTR onSensorPulse() {
  uint32_t now = micros();
  if (isrLastTime_us != 0) {
    isrDelta_us = now - isrLastTime_us;
  }
  isrLastTime_us = now;
  isrTotalPulses++;
  isrNewPulse = true;
}

// ============================================================
// RESET STATE
// ============================================================
void resetAllState() {
  noInterrupts();
  isrLastTime_us = 0;
  isrDelta_us    = 0;
  isrTotalPulses = 0;
  isrNewPulse    = false;
  interrupts();

  smoothedSpeed_kmh = 0.0f;
  topSpeed_kmh      = 0.0f;
  totalDistance_m   = 0.0f;
  lapRunning        = false;
  lapFinished       = false;
  lapTime_s         = 0.0f;
  lapStartTime_us   = 0;

  for (uint8_t i = 0; i < NUM_MILESTONES; i++) {
    milestones[i] = {false, 0.0f, 0.0f};
  }
}

// ============================================================
// HTTP ROUTES
// AsyncWebServer 3.x:
//   * send_P() -> send() dengan FPSTR() untuk PROGMEM string
//   * API handler lambda signature tidak berubah
//   * FastLED.show() DILARANG dipanggil di sini -- set flag saja
// ============================================================
void setupRoutes() {

  // GET / -- halaman utama
  // PROGMEM string: gunakan FPSTR() bukan send_P()
  server.on("/", HTTP_GET, [](AsyncWebServerRequest* req) {
    req->send(200, "text/html", FPSTR(INDEX_HTML));
  });

  // GET /data -- JSON telemetri
  server.on("/data", HTTP_GET, [](AsyncWebServerRequest* req) {
    // Snapshot volatile
    noInterrupts();
    float dist = totalDistance_m;
    float spd  = smoothedSpeed_kmh;
    float top  = topSpeed_kmh;
    bool  run  = lapRunning;
    bool  fin  = lapFinished;
    float lt   = lapTime_s;
    interrupts();

    // Build milestone JSON array
    // Format: [{"hit":true/false,"time_s":X.XXX,"topSpeed":X.X}, ...]
    char msBuf[256];
    int  mLen = 0;
    mLen += snprintf(msBuf + mLen, sizeof(msBuf) - mLen, "[");
    for (uint8_t i = 0; i < NUM_MILESTONES; i++) {
      mLen += snprintf(msBuf + mLen, sizeof(msBuf) - mLen,
        "{\"hit\":%s,\"time_s\":%.3f,\"topSpeed\":%.1f}%s",
        milestones[i].hit ? "true" : "false",
        milestones[i].time_s,
        milestones[i].topSpeed_kmh,
        (i < NUM_MILESTONES - 1) ? "," : "");
    }
    mLen += snprintf(msBuf + mLen, sizeof(msBuf) - mLen, "]");

    char buf[380];
    snprintf(buf, sizeof(buf),
      "{\"speed\":%.2f,\"topSpeed\":%.2f,\"lapTime\":%.3f,"
      "\"distance\":%.2f,\"lapRunning\":%s,\"lapFinished\":%s,"
      "\"milestones\":%s}",
      spd, top, lt, dist,
      run ? "true" : "false",
      fin ? "true" : "false",
      msBuf);

    req->send(200, "application/json", buf);
  });

  // GET /start
  server.on("/start", HTTP_GET, [](AsyncWebServerRequest* req) {
    resetAllState();
    lapRunning      = true;
    lapStartTime_us = micros();
    req->send(200, "text/plain", "OK");
  });

  // GET /reset
  server.on("/reset", HTTP_GET, [](AsyncWebServerRequest* req) {
    resetAllState();
    req->send(200, "text/plain", "OK");
  });

  // GET /ledmode?m=0..5
  // PENTING: tidak ada FastLED.show() di sini -- set flag saja
  server.on("/ledmode", HTTP_GET, [](AsyncWebServerRequest* req) {
    if (req->hasParam("m")) {
      int m = req->getParam("m")->value().toInt();
      if (m >= 0 && m <= 5) {
        ledMode      = (LedMode)m;
        ledResetAnim = true;   // Loop() akan reset state animasi
        ledNeedsUpdate = true; // Loop() akan panggil FastLED.show()
      }
    }
    req->send(200, "text/plain", "OK");
  });

  // GET /ledcolor?r=&g=&b=
  server.on("/ledcolor", HTTP_GET, [](AsyncWebServerRequest* req) {
    if (req->hasParam("r") && req->hasParam("g") && req->hasParam("b")) {
      solidR = (uint8_t)constrain(req->getParam("r")->value().toInt(), 0, 255);
      solidG = (uint8_t)constrain(req->getParam("g")->value().toInt(), 0, 255);
      solidB = (uint8_t)constrain(req->getParam("b")->value().toInt(), 0, 255);
      ledNeedsUpdate = true;
    }
    req->send(200, "text/plain", "OK");
  });

  // GET /ledset?bright=&speed=
  server.on("/ledset", HTTP_GET, [](AsyncWebServerRequest* req) {
    if (req->hasParam("bright")) {
      ledBrightness  = (uint8_t)constrain(req->getParam("bright")->value().toInt(), 1, 255);
      ledNeedsUpdate = true;
    }
    if (req->hasParam("speed")) {
      ledSpeed = (uint8_t)constrain(req->getParam("speed")->value().toInt(), 1, 100);
    }
    req->send(200, "text/plain", "OK");
  });

  server.onNotFound([](AsyncWebServerRequest* req) {
    req->send(404, "text/plain", "Not Found");
  });
}

// ============================================================
// LED -- UPDATE (hanya dipanggil dari loop())
// animInterval(): speed=1->500ms, speed=100->10ms
// ============================================================
inline uint32_t animInterval() {
  return (uint32_t)map(ledSpeed, 1, 100, 500, 10);
}

// Helper: copy leds1 ke leds2, lalu show kedua strip sekaligus
// FastLED.show() satu kali = update semua controller terdaftar
inline void syncAndShow() {
  memcpy(leds2, leds1, sizeof(CRGB) * NUM_LEDS);
  FastLED.show();
}

void updateLeds() {

  // -- Reset animasi jika diminta (ganti mode dari web) -------
  if (ledResetAnim) {
    ledResetAnim = false;
    ledStep      = 0;
    ledHue       = 0;
    wipeIdx      = 0;
    wipeForward  = true;
    blinkState   = false;
    breathVal    = 0.0f;
    breathDir    = 1.0f;
    fill_solid(leds1, NUM_LEDS, CRGB::Black);
    syncAndShow();
    ledLastMs = millis();
    return;
  }

  // -- Brightness berubah -- terapkan segera -------------------
  if (ledNeedsUpdate) {
    ledNeedsUpdate = false;
    FastLED.setBrightness(ledBrightness);
  }

  // -- Rate limiting animasi -----------------------------------
  uint32_t now = millis();
  uint32_t interval = animInterval();
  if (now - ledLastMs < interval) return;
  ledLastMs = now;

  // -- Baca volatile param sekali ke lokal --------------------
  CRGB solidColor = CRGB(solidR, solidG, solidB);
  LedMode mode    = ledMode;

  // Semua animasi hanya tulis ke leds1
  // syncAndShow() akan copy ke leds2 dan show keduanya
  switch (mode) {

    case MODE_SOLID:
      fill_solid(leds1, NUM_LEDS, solidColor);
      break;

    case MODE_BLINK:
      blinkState = !blinkState;
      fill_solid(leds1, NUM_LEDS, blinkState ? solidColor : CRGB::Black);
      break;

    case MODE_CHASE:
      fill_solid(leds1, NUM_LEDS, CRGB::Black);
      for (int t = 3; t >= 1; t--) {
        int idx = ((int)ledStep - t + NUM_LEDS) % NUM_LEDS;
        leds1[idx] = solidColor;
        leds1[idx].nscale8(55 * t);
      }
      leds1[ledStep % NUM_LEDS] = solidColor;
      ledStep = (ledStep + 1) % NUM_LEDS;
      break;

    case MODE_BREATHING:
      breathVal += breathDir * 12.0f;
      if (breathVal >= 255.0f) { breathVal = 255.0f; breathDir = -1.0f; }
      if (breathVal <=   0.0f) { breathVal =   0.0f; breathDir =  1.0f; }
      { CRGB c = solidColor; c.nscale8((uint8_t)breathVal);
        fill_solid(leds1, NUM_LEDS, c); }
      break;

    case MODE_SINELON:
      fadeToBlackBy(leds1, NUM_LEDS, 20);
      { int pos = beatsin16(13, 0, NUM_LEDS - 1);
        leds1[pos] += CHSV(ledHue, 230, 200);
        ledHue++; }
      break;

    case MODE_COLORWIPE:
      if (wipeForward) {
        leds1[wipeIdx] = CHSV(ledHue, 255, 255);
        wipeIdx++;
        if (wipeIdx >= NUM_LEDS) {
          wipeIdx     = NUM_LEDS - 1;
          wipeForward = false;
          ledHue     += 43;
        }
      } else {
        leds1[wipeIdx] = CRGB::Black;
        if (wipeIdx == 0) { wipeForward = true; }
        else { wipeIdx--; }
      }
      break;
  }

  syncAndShow();   // Aman: hanya dari loop()
}

// ============================================================
// SETUP
// ============================================================
void setup() {
  Serial.begin(115200);
  Serial.println(F("\n+==============================+"));
  Serial.println(F("|   DragMeter v4  -- Boot      |"));
  Serial.println(F("+==============================+"));

  // -- FastLED (2 strip, 1 controller, GPIO 16 & 17) ----------
  FastLED.addLeds<LED_TYPE, LED_PIN_1, COLOR_ORDER>(leds1, NUM_LEDS)
         .setCorrection(TypicalLEDStrip);
  FastLED.addLeds<LED_TYPE, LED_PIN_2, COLOR_ORDER>(leds2, NUM_LEDS)
         .setCorrection(TypicalLEDStrip);
  FastLED.setBrightness(ledBrightness);
  fill_solid(leds1, NUM_LEDS, CRGB::Black);
  fill_solid(leds2, NUM_LEDS, CRGB::Black);
  FastLED.show();
  Serial.printf("[LED]    2x%d LEDs @ GPIO%d+GPIO%d -- OK\n",
    NUM_LEDS, LED_PIN_1, LED_PIN_2);

  // -- Sensor interrupt ---------------------------------------
  pinMode(SENSOR_PIN, INPUT_PULLDOWN);
  attachInterrupt(digitalPinToInterrupt(SENSOR_PIN), onSensorPulse, RISING);
  Serial.printf("[Sensor] GPIO%d RISING -- OK\n", SENSOR_PIN);

  // -- WiFi AP ------------------------------------------------
  WiFi.softAPConfig(AP_IP, AP_GATEWAY, AP_SUBNET);
  WiFi.softAP(AP_SSID, AP_PASSWORD, AP_CHANNEL, AP_HIDDEN, AP_MAX_CLIENTS);
  Serial.printf("[WiFi]   SSID:\"%s\" CH:%d IP:%s\n",
    AP_SSID, AP_CHANNEL, WiFi.softAPIP().toString().c_str());

  // -- Web server ---------------------------------------------
  setupRoutes();
  server.begin();
  Serial.println(F("[Server] http://192.168.4.1 -- Ready"));
  Serial.println(F("--------------------------------------"));

  // -- Reset milestone state -----------------------------------
  for (uint8_t i = 0; i < NUM_MILESTONES; i++) {
    milestones[i] = {false, 0.0f, 0.0f};
  }
}

// ============================================================
// LOOP
// * Proses fisika (speed, distance, milestone, lap)
// * Update LED -- FastLED.show() HANYA di sini
// * Tidak ada server.handleClient() -- AsyncWebServer auto
// ============================================================
void loop() {

  // -- Snapshot volatile ISR -----------------------------------
  noInterrupts();
  bool     hasNew = isrNewPulse;
  uint32_t delta  = isrDelta_us;
  uint32_t lastT  = isrLastTime_us;
  uint32_t pulses = isrTotalPulses;
  isrNewPulse     = false;
  interrupts();

  uint32_t now_us = micros();

  // -- Timeout 500ms -> speed = 0 ------------------------------
  if (lastT != 0 && (now_us - lastT) > NO_PULSE_TIMEOUT_US) {
    smoothedSpeed_kmh = 0.0f;
    noInterrupts();
    isrLastTime_us = 0;
    interrupts();
  }

  // -- Hitung kecepatan ----------------------------------------
  if (hasNew && delta > 0) {
    float distPP  = WHEEL_CIRC_M / (float)PULSES_PER_REV;
    float spd_ms  = distPP / (delta * 1e-6f);
    float spd_kmh = spd_ms * 3.6f;

    smoothedSpeed_kmh = (SMOOTHING_ALPHA * spd_kmh)
                      + ((1.0f - SMOOTHING_ALPHA) * smoothedSpeed_kmh);

    if (smoothedSpeed_kmh > topSpeed_kmh)
      topSpeed_kmh = smoothedSpeed_kmh;
  }

  // -- Hitung jarak --------------------------------------------
  totalDistance_m = ((float)pulses / (float)PULSES_PER_REV) * WHEEL_CIRC_M;

  // -- Cek milestone & lap -------------------------------------
  if (lapRunning && !lapFinished) {
    uint32_t elapsed_us = micros() - lapStartTime_us;
    lapTime_s = elapsed_us * 1e-6f;

    for (uint8_t i = 0; i < NUM_MILESTONES; i++) {
      if (!milestones[i].hit && totalDistance_m >= MILESTONES_M[i]) {
        milestones[i].hit          = true;
        milestones[i].time_s       = lapTime_s;
        milestones[i].topSpeed_kmh = topSpeed_kmh;

        Serial.printf("[MS %d]   %.0fm -- %.3fs | Top: %.1f km/h\n",
          i, MILESTONES_M[i], lapTime_s, topSpeed_kmh);

        // Milestone terakhir = finish
        if (i == NUM_MILESTONES - 1) {
          lapFinished = true;
          lapRunning  = false;
          Serial.printf("[FINISH] %.3f s | Top: %.1f km/h\n",
            lapTime_s, topSpeed_kmh);
        }
      }
    }
  }

  // -- LED (FastLED.show() hanya dari sini) -------------------
  updateLeds();

  // yield() -- beri waktu FreeRTOS untuk task WiFi/async
  yield();
}
